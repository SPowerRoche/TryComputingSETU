# Solution

~~~java
float x = 0;

void setup() {
  size(600, 400);
  fill(100, 200, 255);
}

void draw() {
  background(20, 20, 30);
  x = (x + 2) % width;
  ellipse(x, height/2, 80, 80);
  if (keyPressed) {
    if (key == 'r') 
      fill(255, 0, 0);
    else if (key == 'g') 
      fill(0, 255, 0);
    else if (key == 'b') 
      fill(0, 0, 255);
    else if (key == 'm') 
      fill(random(255), random(255), random(255));
  }
}
~~~
