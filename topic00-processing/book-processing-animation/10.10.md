# Solutions

## Drawing a Scene
~~~java
size(600, 400);

background(135, 206, 235);

// --- Ground ---
fill(60, 180, 75);
noStroke();
rect(0, 300, 600, 100);

// --- Sun ---
fill(255, 204, 0);
circle(500, 80, 80);

// --- House base ---
fill(200, 100, 60);
rect(200, 200, 200, 120);

// --- Roof ---
fill(150, 50, 30);
triangle(180, 200, 420, 200, 300, 130);

// --- Door ---
fill(100, 60, 40);
rect(270, 240, 60, 80);

// --- Windows ---
fill(255);
rect(220, 220, 40, 40);
rect(340, 220, 40, 40);

// --- Tree trunk ---
fill(102, 51, 0);
rect(80, 230, 30, 90);

// --- Tree leaves ---
fill(34, 139, 34);
ellipse(95, 210, 120, 120);
~~~

## Change Colour

~~~java

float sunX = -50;   // start just off the left side
boolean grassToggle = false;
color grassColor = color(60, 180, 75);

void setup() {
  size(600, 400);
}

void draw() {
  // 1. Draw background FIRST (sequence!)
  background(135, 206, 235);  // sky blue

  // 2. Draw ground SECOND
  fill(grassColor);
  rect(0, 300, width, 100);

  // 3. Update the sunâ€™s position BEFORE drawing
  sunX = (sunX + 2) % width;

  // 4. Draw the sun LAST so it appears on top
  fill(255, 204, 0);
  ellipse(sunX, 50, 80, 80);
  
  // 5. Draw the house
  // House body
  fill(200, 100, 50);
  rect(150, 200, 150, 120);

  // Roof
  fill(150, 50, 30);
  triangle(150, 200, 225, 140, 300, 200);

  // Door
  fill(100, 60, 30);
  rect(215, 250, 30, 70);

  // Window
  fill(180, 230, 255);
  rect(170, 230, 40, 40);

  // 6. Draw the tree LAST (in front of house)
  // Trunk
  fill(120, 70, 20);
  rect(400, 240, 30, 80);

  // Foliage
  fill(20, 150, 50);
  ellipse(415, 220, 100, 100);
}

void mousePressed(){
  // 2. Change grass colour if clicked on ground
  if (mouseY > 300) {
    grassToggle = !grassToggle;
    if (grassToggle) {
      grassColor = color(40, 160, 60);  // darker green
    } else {
      grassColor = color(60, 180, 75);  // original
    }
    return;
  }
}
