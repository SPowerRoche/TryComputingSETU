# Solution

## Change Colour

~~~java
float x = 0;

void setup() {
  size(600, 400);
  fill(100, 200, 255);
}

void draw() {
  background(20, 20, 30);
  x = (x + 2) % width;
  ellipse(x, height/2, 80, 80);
  if (keyPressed) {
    if (key == 'r') 
      fill(255, 0, 0);
    else if (key == 'g') 
      fill(0, 255, 0);
    else if (key == 'b') 
      fill(0, 0, 255);
    else if (key == 'm') 
      fill(random(255), random(255), random(255));
  }
}
~~~

## Drawing the House
~~~java
float sunX = -50;   // start just off the left side

void setup() {
  size(600, 400);
}

void draw() {
  // 1. Draw background FIRST (sequence!)
  background(135, 206, 235);  // sky blue

  // 2. Draw ground SECOND
  fill(50, 200, 70);
  rect(0, 300, width, 100);

  // 3. Update the sunâ€™s position BEFORE drawing
  sunX = (sunX + 2) % width;

  // 4. Draw the sun LAST so it appears on top
  fill(255, 204, 0);
  ellipse(sunX, 50, 80, 80);
  
  // 5. Draw the house
  // House body
  fill(200, 100, 50);
  rect(150, 200, 150, 120);

  // Roof
  fill(150, 50, 30);
  triangle(150, 200, 225, 140, 300, 200);

  // Door
  fill(100, 60, 30);
  rect(215, 250, 30, 70);

  // Window
  fill(180, 230, 255);
  rect(170, 230, 40, 40);

  // 6. Draw the tree LAST (in front of house)
  // Trunk
  fill(120, 70, 20);
  rect(400, 240, 30, 80);

  // Foliage
  fill(20, 150, 50);
  ellipse(415, 220, 100, 100);
}
~~~
